For each of the following valid arguments, complete a Fitch-style natural deduction to prove its validity. Do this by filling in the missing lines, citations, or both.

\problems
\problem{1}
\argument
 A ^ B
 C ^ D
\argumentline
 A ^ D
\endargument
	\answer
	\fitchproof
	 1. | A ^ B
	 2. | C ^ D  ..  Goal: A ^ D
	    |---
	 3. | A      .  ^ Elim: 1
	 4. | B      .  ^ Elim: 1  .  (optional)
	 5. |        .  ^ Elim: 2  .  (optional)
	 6. |        .  ^ Elim: 2
	 7. | A ^ D  .  ^ Intro: 3, 6
	\endfitchproof
	\endanswer

\problem{6}
\argument
 F(a) ^ a=b
 G(b)
\argumentline
 F(b) ^ G(a)
\endargument
	\answer
	\fitchproof
	 1. | F(a) ^ a=b
	 2. | G(b)         ..  Goal: F(b) ^ G(a)
	    |---
	 3. | F(a)         .  ^ Elim: 1
	 4. | a=b          .  ^ Elim: 1
	 5. | F(b)         .  = Elim: 3, 4
	 6. |              .  = Elim: 2, 4
	 7. |              .  ^ Intro: 5, 6
	\endfitchproof
	\endanswer
