For each of the following valid arguments, complete a Fitch-style natural deduction to prove its validity. Do this by filling in the missing lines, citations, or both.

\problems
\problem{1}
\argument
 A ^ B
 B > C
\argumentline
 C v D
\endargument
	\answer
	\fitchproof
	 1. | A ^ B
	 2. | B > C  ..  Goal: C v D
	    |---
	 3. | B      .  ^ Elim: 1
	 4. |        .  > Elim: 2, 3
	 5. |        .  v Intro: 4
	\endfitchproof
	\endanswer

\problem{3}
\argument
 A > B
 B > C
\argumentline
 A > [C v D]
\endargument
	\answer
	\fitchproof
	 1. | A > B
	 2. | B > C         ..  Goal: A > [C v D]
	    |---
	 3. |  | A          ..  Assumption  .  Setting up: > Intro  .  Goal: C v D
	    |  |---
	 4. |  | 
	\endfitchproof
	\endanswer

\problem{5}
\argument
 A > C
 B > C
 [D ^ A] v [E ^ B]
\argumentline
 C
\endargument
	\answer
	\fitchproof
	 1. | A > C
	 2. | B > C
	 3. | [D ^ A] v [E ^ B]  ..  Goal: C
	    |---
	 4. |  | D ^ A  ..  Assumption  .  Setting up: v Elim: 3  .  Goal: C
	    |  |---
	    |  | 
	    |  ++++
	    |  +
	 ?. |  | E ^ B  ..  Assumption  .  Setting up: v Elim: 3  .  Goal: C
	    |  |---
	    |  | 
	    |  ++++
	    | C         .  v Elim: 3, 4-, ?-
	\endfitchproof
	\endanswer

\problem{6}
\argument
 A > C
 B > D
\argumentline
 [A v B] > [C v D]
\endargument
	\answer
	\fitchproof
	 1. | A > C
	 2. | B > D        ..  Goal: [A v B] > [C v D]
	    |---
	 3. |  | A v B     ..  Setting up: > Intro    .  Goal: C v D
	    |  |---
	 4. |  | 
	\endfitchproof
	\endanswer
